SELECT [KODE_CUST] as doctor_nu
      ,[NAMA_CUST] as name
      ,[ALMT_CUST] as address
      ,[ALMT_CUST2] as address2
      ,[KODE_MDRP1]
FROM (
  SELECT *,
    ROW_NUMBER() OVER (
      PARTITION BY [KODE_CUST]
      ORDER BY [KODE_CUST] ASC, [NAMA_CUST] ASC
    ) AS rn
  FROM [T_DOKTER]
) t
WHERE rn = 1;
