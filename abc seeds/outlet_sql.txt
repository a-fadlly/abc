SELECT kd_outlet as outlet_nu,
nm_outlet as name,
alamat_outlet as address,
kd_outlet_uni as outlet_nu_uni,
nm_outlet_uni as name_uni,
alamat_outlet_uni as address_uni,
GETDATE() as created_at,
GETDATE() as updated_at
FROM (
  SELECT *,
    ROW_NUMBER() OVER (
      PARTITION BY kd_outlet
      ORDER BY kd_outlet ASC, nm_outlet_uni ASC
    ) AS rn
  FROM [T_M_Outlet]
) t
WHERE rn = 1;
